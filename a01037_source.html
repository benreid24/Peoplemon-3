290 bool PeoplemonInfo::observe(const bl::input::Actor&, unsigned int cmd, bl::input::DispatchType,
291  bool eventTriggered) {
292  if (inputTimer.getElapsedTime().asSeconds() < 0.4f && !eventTriggered) return true;
293  inputTimer.restart();
294 
295  switch (cmd) {
296  case core::input::Control::MoveLeft:
297  if (activePage == ActivePage::Moves) { setPage(ActivePage::Basics); }
298  break;
299  case core::input::Control::MoveRight:
300  if (activePage == ActivePage::Basics) { setPage(ActivePage::Moves); }
301  break;
302  case core::input::Control::MoveDown:
303  if (activePage == ActivePage::Moves) {
304  moveMenu.processEvent(
305  bl::menu::Event(bl::menu::Event::MoveEvent(bl::menu::Item::AttachPoint::Bottom)));
306  }
307  break;
308  case core::input::Control::MoveUp:
309  if (activePage == ActivePage::Moves) {
310  moveMenu.processEvent(
311  bl::menu::Event(bl::menu::Event::MoveEvent(bl::menu::Item::AttachPoint::Top)));
312  }
313  break;
314  case core::input::Control::Back:
315  systems.engine().popState();
316  break;
317  default:
318  break;
319  }
320 
321  return true;
322 }
