212 bool NpcEditorWindow::validate(bool saving) const {
213  if (saving) {
214  if (fileLabel->getText() == EmptyFile) {
215  bl::dialog::tinyfd_messageBox("Warning", "Please select a file", "ok", "warning", 1);
216  return false;
217  }
218  }
219  else {
220  if (!FileUtil::exists(
221  FileUtil::joinPath(core::Properties::NpcPath(), fileLabel->getText()))) {
222  bl::dialog::tinyfd_messageBox("Warning", "Bad file selected", "ok", "warning", 1);
223  return false;
224  }
225  }
226  if (nameEntry->getInput().empty()) {
227  bl::dialog::tinyfd_messageBox("Warning", "Enter a name", "ok", "warning", 1);
228  return false;
229  }
230  if (!FileUtil::directoryExists(
231  FileUtil::joinPath(core::Properties::CharacterAnimationPath(), animLabel->getText()))) {
232  bl::dialog::tinyfd_messageBox("Warning", "Bad animation", "ok", "warning", 1);
233  return false;
234  }
235  const std::string p =
236  FileUtil::joinPath(core::Properties::ConversationPath(), convLabel->getText());
237  if (!FileUtil::exists(p)) {
238  bl::dialog::tinyfd_messageBox("Warning", "Bad conversation", "ok", "warning", 1);
239  return false;
240  }
241  if (behaviorEditor.getValue().type() == core::file::Behavior::Type::FollowingPath &&
242  behaviorEditor.getValue().path().paces.empty()) {
243  bl::dialog::tinyfd_messageBox("Warning", "Behavior path is empty", "ok", "warning", 1);
244  return false;
245  }
246  return true;
247 }
