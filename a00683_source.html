131 void LightingSystem::update(float dt) {
132  if (weatherModifier != targetWeatherModifier) {
133  weatherResidual += AdjustSpeed * dt;
134  const float p = std::floor(weatherResidual);
135  if (p > 0.f) {
136  weatherResidual -= p;

146  weatherModifier = targetWeatherModifier;

185 void LightingSystem::observe(const event::TimeChange& now) {
186  if (adjustsForSunlight()) {
187  const float x = now.newTime.hour * 60 + now.newTime.minute;
188  const float n = 0.7f;
189  sunlightFactor = 1.f - (0.5 * std::cos(3.1415926 * x / 720) + 0.5) *
190  ((1 - n) * (720 - x) * (720 - x) / 518400 + n);
191  }
192  else { sunlightFactor = 1.f; }
