247 
248  const auto output = [&n, i](const OutputCb& cb, const std::string& m) {
249  std::string o = "Node " + std::to_string(i) + ": " + m;
250  for (unsigned int i = 0; i < o.size(); ++i) {
251  if (o[i] == '\'') o[i] = '`';
252  if (o[i] == '"') {
253  o.erase(i, 1);
254  o.insert(i, "<QUOTE>");
255  }
256  }
257  cb(o);
258  };
259 
260  switch (n.getType()) {
261  case T::Talk:
262  case T::Prompt:
263  if (n.message().empty()) { output(warning, "Message is empty"); }
264  if (n.getType() == T::Prompt) {
265  if (n.choices().empty()) {
266  output(error, "Prompt has no choices");
267  return false;
268  }
269  for (const auto& c : n.choices()) {
270  if (c.first.empty()) {
271  output(error, "Choice has empty text");
272  return false;
273  }
274  }
275  }
276  return true;
277 
278  case T::TakeItem:
279  case T::GiveItem:
280  if (n.item().id == core::item::Id::Unknown) {
281  output(error, "Invalid item");
282  return false;
283  }
284  return true;
285 
286  case T::GiveMoney:
287  case T::TakeMoney:
288  if (n.money() == 0) {
289  output(error, "Invalid money amount");
290  return false;
291  }
292  return true;
293 
294  case T::RunScript: {
295  bl::script::Script check(n.script());
296  if (!check.valid()) {
297  output(error, "SyntaxError: " + check.errorMessage());
298  return false;
299  }
300  }
301  return true;
302 
303  case T::SetSaveFlag:
304  case T::CheckSaveFlag:
305  if (n.saveFlag().empty()) {
306  output(error, "Save flag cannot be empty");
307  return false;
308  }
309  return true;
310 
311  case T::CheckInteracted:
312  return true;
313 
314  default:
315  output(error, "Unknown node type");
316  return false;
317  }
