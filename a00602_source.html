56  auto set = lights.getCellAndNeighbors(position.x, position.y);
57  Handle closest = None;
58  unsigned long cdist = 10000000;

61  const long dy = light.get().second.position.y - position.y;

207 void LightingSystem::observe(const event::TimeChange& now) {
