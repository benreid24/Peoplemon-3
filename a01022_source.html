174 void TrainerSequence::render(sf::RenderTarget& target, float lag) {
175  // compute progress and positions
176  const float t = time + lag;
177  const float progress = t / IntroLength;
178  const sf::View& view = target.getView();
179  const sf::Vector2f scorner = view.getCenter() - view.getSize() * 0.5f;
180  const sf::Vector2f ecorner = scorner + view.getSize();
181 
182  // render bars
183  const float bw = view.getSize().x * 0.5f * progress * 1.15f;
184  for (float y = scorner.y + barCircle.getRadius(); y <= ecorner.y - barCircle.getRadius();
185  y += barCircle.getRadius() * 2.f) {
186  for (float x = 0.f; x <= bw; x += 1.f) {
187  barCircle.setPosition(scorner.x + x, y);
188  target.draw(barCircle);
189  barCircle.setPosition(ecorner.x - x, y);
190  target.draw(barCircle);
191  }
192  }
193 
194  // render circle
195  ball.setScale(progress, progress);
196  target.draw(ball);
197 }
