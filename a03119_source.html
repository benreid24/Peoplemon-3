130 unsigned int Peoplemon::levelUpXp(Id id, unsigned int level) {
131  const auto it = xpGroups->find(id);
132  if (it == xpGroups->end()) return std::numeric_limits<unsigned int>::max();
133  switch (it->second) {
134  case 0:
135  return 5 * level * level * level / 4;
136  case 1:
137  return level * level * level;
138  case 2:
139  return 4 * level * level * level / 5;
140  default:
141  BL_LOG_ERROR << "Invalid xp group " << it->second << " for peoplemon "
142  << static_cast<unsigned int>(id);
143  return std::numeric_limits<unsigned int>::max();
144  }
