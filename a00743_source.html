147 void HUD::printDoneStateTransition() {
148  if (queuedOutput.front().getType() == Item::Message) { state = WaitingContinue; }
149  else if (queuedOutput.front().getType() == Item::Prompt) {
150  state = WaitingPrompt;
151  const std::vector<std::string>& choices = queuedOutput.front().getChoices();
152 
153  bl::menu::Item::Ptr mitem =
154  bl::menu::TextItem::create(choices.empty() ? "INVALID" : choices.front(),
155  Properties::MenuFont(),
156  sf::Color::Black,
157  Properties::HudFontSize());
158  mitem->getSignal(bl::menu::Item::Activated)
159  .willAlwaysCall(std::bind(&HUD::choiceMade, this, 0));
160  choiceMenu.setRootItem(mitem);
161 
162  bl::menu::Item* prev = mitem.get();
163  for (unsigned int i = 1; i < choices.size(); ++i) {
164  mitem = bl::menu::TextItem::create(
165  choices[i], Properties::MenuFont(), sf::Color::Black, Properties::HudFontSize());
166  mitem->getSignal(bl::menu::Item::Activated)
167  .willAlwaysCall(std::bind(&HUD::choiceMade, this, i));
168  choiceMenu.addItem(mitem, prev, bl::menu::Item::Bottom);
169  prev = mitem.get();
170  }
171  const sf::FloatRect bounds = choiceMenu.getBounds();
172  const float y = viewSize.y - bounds.height - 18.f;
173  choiceMenu.setPosition({choiceBoxX + 18.f, y + 2.f});
174  choiceDriver.drive(&choiceMenu);
175  }
176  else if (queuedOutput.front().getType() == Item::Keyboard) {
177  state = WaitingKeyboard;
178  screenKeyboard.start(queuedOutput.front().minInputLength(),
179  queuedOutput.front().maxInputLength());

207 void HUD::HudListener::process(component::Command cmd) {
208  switch (owner.state) {
209  case Printing:
210  if (isNextCommand(cmd)) {
211  owner.currentMessage.showAll();
212  owner.displayText.setString(owner.currentMessage.getContent());
213  owner.printDoneStateTransition();
214  }
215  break;
216 
217  case WaitingContinue:
218  if (isNextCommand(cmd)) {
219  owner.queuedOutput.front().getCallback()(owner.queuedOutput.front().getMessage());
220  owner.next();
221  }
222  break;
223 
224  case WaitingPrompt:
225  owner.choiceDriver.process(cmd);
226  break;
227 
228  case WaitingKeyboard:
229  owner.screenKeyboard.process(cmd);
230  break;
231 
232  default:
233  BL_LOG_WARN << "Input received by HUD while in invalid state: " << owner.state;
234  break;
235  }
236 }

285 void HUD::EntryCard::update(float dt) {
286  constexpr float MoveSpeed = 150.f;
287  constexpr float HoldTime = 2.5f;
288 
289  switch (state) {
290  case Dropping:
291  stateVar -= MoveSpeed * dt;
292  if (stateVar <= 0.f) {
293  stateVar = 0.f;
294  state = Holding;
295  }
296  break;
297 
298  case Rising:
299  stateVar += MoveSpeed * dt;
300  if (stateVar >= card.getGlobalBounds().height) { state = Hidden; }
301  break;
302 
303  case Holding:
304  stateVar += dt;
305  if (stateVar >= HoldTime) {
306  state = Rising;
307  stateVar = 0.f;
308  }
309  break;
310 
311  case Hidden:
312  default:
313  break;
314  }
