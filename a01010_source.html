94 void Evolution::update(bl::engine::Engine&, float dt, float) {
95  const auto updateSpark = [dt](Spark& spark) {
96  spark.time += dt;
97  spark.position += spark.velocity * dt;
98  return spark.time < spark.lifetime;
99  };
100 
101  switch (state) {
102  case AnimState::OldFadeOut:
103  fadeColor = std::max(0.f, fadeColor - FadeRate * dt);
104  oldThumb.setColor(makeFade(fadeColor));
105  if (fadeColor == 0.f) {
106  state = AnimState::SizeOscillating;
107  oscillateTime = 0.f;
108  newThumb.setColor(oldThumb.getColor());
109  newThumb.setScale(0.f, 0.f);
110  }
111  break;
112 
113  case AnimState::SizeOscillating: {
114  oscillateTime = std::min(oscillateTime + dt, ScalingTime);
115  const float s = std::abs(bl::math::sin(SinMult * oscillateTime * oscillateTime));
116  const float os = 1.f - s;
117  oldThumb.setScale(os, os);
118  newThumb.setScale(s, s);
119  if (oscillateTime >= ScalingTime) {
120  state = AnimState::NewFadeIn;
121  newThumb.setScale(1.f, 1.f);
122  fadeColor = 0.f;
123  sparks.setTargetCount(SparkCount);
124  sparks.setCreateRate(SparkSpawnRate);
125  }
126  } break;
127 
128  case AnimState::NewFadeIn:
129  fadeColor = std::min(255.f, fadeColor + FadeRate * dt);
130  newThumb.setColor(makeFade(fadeColor));
131  sparks.update(updateSpark, dt);
132  if (fadeColor >= SparkStopColor) {
133  sparks.setTargetCount(0);
134  sparks.setCreateRate(0.f);
135  }
136  if (fadeColor >= 255.f) {
137  state = AnimState::EvolvedMsg;
138  ppl.evolve();
139  systems.hud().dismissStickyMessage();
140  systems.hud().displayMessage(ppl.name() + " evolved into " +
141  core::pplmn::Peoplemon::name(ppl.evolvesInto()) + "!",
142  std::bind(&Evolution::messageDone, this));
143  }
144  break;
145  case AnimState::EvolvedMsg:
146  sparks.update(updateSpark, dt);
147  [[fallthrough]];
148 
149  case AnimState::IntroMsg:
150  case AnimState::CancelMsg:
151  case AnimState::CancelConfirm:
152  systems.hud().update(dt);
153  break;
154 
155  default:
156  break;
157  }
158 }
