802 void waitUntilTime(system::Systems& systems, SymbolTable& table, const std::vector<Value>& args,
803  Value&) {
804  Value::validateArgs<PrimitiveValue::TAny, PrimitiveValue::TBool>("waitUntilTime", args);
805 
806  const system::Clock::Time now = systems.clock().now();
807  const system::Clock::Time then = parseTime(args[0]);
808  const unsigned int nts = now.hour * 60 + now.minute;
809  const unsigned int ets = then.hour * 60 + then.minute;
810  if (nts < ets || (nts != ets && args[1].value().getAsBool())) {
811  bl::util::Waiter waiter;
812  const auto unlock = [&waiter]() { waiter.unblock(); };
813  ClockTrigger trigger(unlock, then);
814  bl::event::ClassGuard<event::TimeChange> guard(&trigger);
815  guard.subscribe(systems.engine().eventBus());
816  table.waitOn(waiter);
817  }
818 }
