136 void StatBoxes::update(float dt) {
137  const float change = dt * BarRate * BarSize.x;
138 
139  // opponent box slide
140  if (oState == State::Sliding) {
141  const float dx = SlideSpeed * dt;
142  opBox.move(dx, 0.f);
143  if (opBox.getPosition().x > OpBoxPos.x) {
144  opBox.setPosition(OpBoxPos);
145  oState = State::Showing;
146  }
147  opHpBar.setPosition(opBox.getPosition() + OpBarPos);
148  opAil.setPosition(opBox.getPosition() + sf::Vector2f(10.f, 46.f));
149  opName.setPosition(opBox.getPosition() + sf::Vector2f(5.f, 18.f));
150  opLevel.setPosition(opBox.getPosition() + sf::Vector2f(168.f, 21.f));
151  }
152  else if (oState == State::Showing) {
153  // opponent bar size
154  if (opHpBar.getSize().x < opHpBarTarget) {
155  const float nw = std::min(opHpBarTarget, opHpBar.getSize().x + change);
156  opHpBar.setSize({nw, BarSize.y});
157  }
158  else if (opHpBar.getSize().x > opHpBarTarget) {
159  const float nw = std::max(opHpBarTarget, opHpBar.getSize().x - change);
160  opHpBar.setSize({nw, BarSize.y});
161  }
162 
163  // opponent bar color
164  const float opHp = opHpBar.getSize().x / BarSize.x;
165  opHpBar.setFillColor(Properties::HPBarColor(opHp));
166  }
167 
168  // player box slide
169  if (pState == State::Sliding) {
170  const float dx = SlideSpeed * dt;
171  lpBox.move(-dx, 0.f);
172  if (lpBox.getPosition().x < LpBoxPos.x) {
173  lpBox.setPosition(LpBoxPos);
174  pState = State::Showing;
175  }
176  lpAil.setPosition(lpBox.getPosition() + sf::Vector2f(33.f, 61.f));
177  lpName.setPosition(lpBox.getPosition() + sf::Vector2f(41.f, 22.f));
178  lpLevel.setPosition(lpBox.getPosition() + sf::Vector2f(218.f, 24.f));
179  lpHp.setPosition(lpBox.getPosition() + sf::Vector2f(131.f, 68.f));
180  lpHpBar.setPosition(lpBox.getPosition() + LpBarPos);
181  lpXpBar.setPosition(lpBox.getPosition() + XpBarPos);
182  }
183  else if (pState == State::Showing) {
184  // player bar size
185  if (lpHpBar.getSize().x < lpHpBarTarget) {
186  const float nw = std::min(lpHpBarTarget, lpHpBar.getSize().x + change);
187  lpHpBar.setSize({nw, BarSize.y});
188  }
189  else if (lpHpBar.getSize().x > lpHpBarTarget) {
190  const float nw = std::max(lpHpBarTarget, lpHpBar.getSize().x - change);
191  lpHpBar.setSize({nw, BarSize.y});
192  }
193 
194  // player bar color
195  const float lphp = lpHpBar.getSize().x / BarSize.x;
196  lpHpBar.setFillColor(Properties::HPBarColor(lphp));
197 
198  // xp bar size
199  const float xpChange = dt * BarRate * XpBarSize.x;
200  if (lpXpBar.getSize().x < lpXpBarTarget) {
201  const float nw = std::min(lpXpBarTarget, lpXpBar.getSize().x + xpChange);
