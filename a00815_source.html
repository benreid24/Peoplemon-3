102 void Flight::update(std::mutex&, float dt, float, float, float) {
103  switch (state) {
104  case State::Rising:
105  riseState.height = std::min(riseState.height + RiseRate * dt, RiseHeight);
106  playerPos->transform->setPosition(playerPos->transform->getGlobalPosition().x,
107  riseState.startY - riseState.height);
108  owner.render().updateShadow(owner.player().player(),
109  riseState.height,
110  ShadowSize - (riseState.height / RiseHeight) * ShadowShrinkage);
111  if (riseState.height >= RiseHeight) {
112  state = State::Rotating;
113  flightDest = destination.getWorldPosition(Properties::PixelsPerTile());
114  flightDest.y -= RiseHeight;
115  unitVelocity = flightDest - playerPos->transform->getGlobalPosition();
116  const float m =
117  std::sqrt(unitVelocity.x * unitVelocity.x + unitVelocity.y * unitVelocity.y);
118  unitVelocity.x /= m;
119  unitVelocity.y /= m;
120  rotateState.angle = 0.f;
121  rotateState.targetAngle =
122  bl::math::radiansToDegrees(
123  std::atan2(flightDest.y - playerPos->transform->getGlobalPosition().y,
124  flightDest.x - playerPos->transform->getGlobalPosition().x)) +
125  90.f;
126  rotateState.rotateRate = (rotateState.targetAngle - rotateState.angle) / RotateTime;
127  }
128  break;
129 
130  case State::Rotating:
131  case State::UnRotating:
132  rotateState.angle += rotateState.rotateRate * dt;
133  playerAnim->setAngle(rotateState.angle);
134  if (std::abs(rotateState.angle - rotateState.targetAngle) < 1.5f) {
135  rotateState.angle = rotateState.targetAngle;
136  playerAnim->setAngle(rotateState.angle);
137 
138  if (state == State::Rotating) {
139  const float pa = rotateState.targetAngle;
140  state = State::Accelerating;
141  flyState.angle = pa; // probably not required but eh
142  flyState.velocity = 0.f;
143  flyState.ogDistSqrd = distSqrd(*playerPos, flightDest);
144  flyState.maxSpeed = std::min(std::sqrt(flyState.ogDistSqrd) / MinFlyTime, MaxSpeed);
145  }
146  else {
147  state = State::Descending;
148  riseState.height = RiseHeight;
149  playerPos->position = destination.position;
150  }
151  }
152  break;
153 
154  case State::Accelerating: {

156  const float cDist = distSqrd(*playerPos, flightDest);
157  const float percent = 1.f - (cDist / flyState.ogDistSqrd) + 0.005f;
158  flyState.velocity =
159  percent <= 0.125f ? percent * 8.f * flyState.maxSpeed : flyState.maxSpeed;
160  if (flyState.velocity >= flyState.maxSpeed) {
161  flyState.velocity = flyState.maxSpeed;
162  state = State::Flying;
163  }
164  if (cameraShake) {
165  cameraShake->setMagnitude(flyState.velocity / flyState.maxSpeed * ShakeMagnitude);
166  cameraShake->setShakesPerSecond(flyState.velocity / flyState.maxSpeed * ShakesPerSec);
167  }
168  } break;
169 
170  case State::Flying: {
171  movePlayer(dt);
172  const float cDist = distSqrd(*playerPos, flightDest);
173  const float percent = 1.f - cDist / flyState.ogDistSqrd;
174  if (percent >= 0.98f) { state = State::Deccelerating; }
175  } break;
176 
177  case State::Deccelerating: {
178  movePlayer(dt);
179  const float cDist = distSqrd(*playerPos, flightDest);
180  const float percent = 1.f - cDist / flyState.ogDistSqrd;
181  flyState.velocity = std::max((1.f - percent) * 50.f * flyState.maxSpeed, 32.f * 6.f);
182  if (cDist <= 3.4f || flyState.velocity == 0.f) {
183  if (cameraShake) { cameraShake->setMagnitude(0.f); }
184  playerPos->transform->setPosition(flightDest);
185  const float pa = flyState.angle;
186  state = State::UnRotating;
187  rotateState.angle = pa;
188  rotateState.targetAngle = 0.f;
189  rotateState.rotateRate = (rotateState.targetAngle - rotateState.angle) / RotateTime;
190  }
191  else if (cameraShake) {
192  cameraShake->setMagnitude(flyState.velocity / flyState.maxSpeed * ShakeMagnitude);
193  cameraShake->setShakesPerSecond(flyState.velocity / flyState.maxSpeed * ShakesPerSec);
194  }
195  } break;
196 
197  case State::Descending:
198  riseState.height = std::max(riseState.height - RiseRate * dt, 0.f);
199  playerPos->transform->setPosition(

201  destination.getWorldPosition(Properties::PixelsPerTile()).y - riseState.height);
202  owner.render().updateShadow(owner.player().player(),
203  riseState.height,
