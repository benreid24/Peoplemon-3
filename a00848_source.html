53 void renderArrow(bl::gfx::VertexBuffer& buffer, unsigned int offset, const sf::Vector2f& start,
54  const sf::Vector2f& end) {
55  const float forwardAngle = std::atan2(end.y - start.y, end.x - start.x);
56  const float reverseAngle = std::atan2(start.y - end.y, start.x - end.x);
57  const float perpAngle = (forwardAngle + reverseAngle) * 0.5f;
58 
59  const sf::Vector2f from(start.x + NodeRadius * std::cos(forwardAngle),
60  start.y + NodeRadius * std::sin(forwardAngle));
61  const sf::Vector2f to(end.x + (NodeRadius + ArrowDepth) * std::cos(reverseAngle),
62  end.y + (NodeRadius + ArrowDepth) * std::sin(reverseAngle));
63 
64  const sf::Vector2f sCorner1(from.x + ArrowWidth * std::cos(perpAngle),
65  from.y + ArrowWidth * std::sin(perpAngle));
66  const sf::Vector2f sCorner2(from.x - ArrowWidth * std::cos(perpAngle),
67  from.y - ArrowWidth * std::sin(perpAngle));
68  const sf::Vector2f eCorner1(to.x + ArrowWidth * std::cos(perpAngle),
69  to.y + ArrowWidth * std::sin(perpAngle));

71  to.y - ArrowWidth * std::sin(perpAngle));

73  // Rect triangle 1
74  buffer[offset].position = sCorner1;

76  buffer[offset + 2].position = eCorner2;
77

84  buffer[offset + 6].position = {end.x + NodeRadius * std::cos(reverseAngle),

200 void ConversationTree::update(const std::vector<core::file::Conversation::Node>& nodes) {
201  // Clear existing render data
202  renderNodes.clear();
203  vertexBuffer.resize(0);
204  if (nodes.empty()) return;
205 
206  // Lots of data structures
207  std::queue<unsigned int> toVisit;
208  std::vector<bool> visited(nodes.size(), false);
209  std::vector<std::pair<unsigned int, unsigned int>> edges;
210  std::vector<unsigned int> levelCounts(nodes.size(), 0);
211  std::vector<unsigned int> nodeLevels(nodes.size(), 0);
212  std::vector<bool> terminators(nodes.size(), false);
213 
214  // Initial setup
215  edges.reserve(nodes.size() * 2);
216  toVisit.push(0);
217  visited[0] = true;
218  levelCounts[0] = 1;
219 
220  // BFS through tree to populate data structures
221  std::vector<unsigned int> next;
222  next.reserve(8);
223  while (!toVisit.empty()) {
224  const unsigned i = toVisit.front();
225  const unsigned int dist = nodeLevels[i];
226  const unsigned int newDist = dist + 1;
227  const core::file::Conversation::Node& node = nodes[i];
228  toVisit.pop();
229 
230  core::file::Conversation::getNextJumps(node, next);
231 
232  for (unsigned int jump : next) {
233  if (jump >= nodes.size()) {
234  terminators[i] = true;
235  continue;
236  }
237  edges.emplace_back(i, jump);
238  if (visited[jump]) continue;
239 
240  visited[jump] = true;
241  levelCounts[newDist] += 1;
242  nodeLevels[jump] = newDist;
243  toVisit.push(jump);
244  }
245  }
246 
247  // Allocate space for render data
248  vertexBuffer.resize(edges.size() * ArrowVertexCount + nodes.size() * CircleVertexCount);
249  renderNodes.resize(nodes.size());
250  unsigned int vIndex = 0;
251 
252  // Get ready to position nodes
253  std::vector<sf::Vector2f> nodePositions(nodes.size());
254  std::vector<float> levelLefts(levelCounts.size());
255  for (unsigned int i = 0; i < levelCounts.size(); ++i) {
256  const float width = static_cast<float>(levelCounts[i]) * Spacing;
257  levelLefts[i] = -width * 0.5f;
258  }
259  const unsigned int maxLevelCount =
260  *std::max_element(std::begin(levelCounts), std::end(levelCounts));
261  const float maxwidth = static_cast<float>(maxLevelCount) * Spacing;
262  const float maxLeft = -maxwidth * 0.5f;
263 
264  // Position nodes
265  sf::Vector2f unreachablePos(maxLeft, -Spacing);
266  for (unsigned int i = 0; i < nodes.size(); ++i) {
267  if (visited[i]) {
268  nodePositions[i].x = levelLefts[nodeLevels[i]];
269  nodePositions[i].y = static_cast<float>(nodeLevels[i]) * Spacing;
270  levelLefts[nodeLevels[i]] += Spacing;
271  }
272  else {
273  nodePositions[i] = unreachablePos;
274  unreachablePos.x += NodePadding + NodeRadius * 2.f;
275  }
276  }
277 
278  // Render edges
279  for (const auto& edge : edges) {
280  renderArrow(vertexBuffer, vIndex, nodePositions[edge.first], nodePositions[edge.second]);
281  vIndex += ArrowVertexCount;
282  }
283 
284  // Render circles and text
285  for (unsigned int i = 0; i < nodes.size(); ++i) {
286  renderCircle(vertexBuffer, vIndex, nodePositions[i], terminators[i]);
287  renderNodes[i].setup(
288  nodePositions[i], i, nodes[i], vIndex, vIndex + CircleVertexCount, terminators[i]);
289  vIndex += CircleVertexCount;
290  }
291 
292  // Reset bounds
293  centerView();
294 
295  vertexBuffer.update();
296 }
