741  result.create(size.x * core::Properties::PixelsPerTile(),
742  size.y * core::Properties::PixelsPerTile());
743  sf::RenderTexture patch;
744  patch.create(PatchSize * core::Properties::PixelsPerTile(),
745  PatchSize * core::Properties::PixelsPerTile());
746 
747  int pw = size.x / PatchSize;
748  if (size.x % PatchSize > 0) ++pw;
749  int ph = size.y / PatchSize;
750  if (size.y % PatchSize > 0) ++ph;
751  BL_LOG_DEBUG << "Performing render in " << (pw * ph) << " patches";
752 
753  const std::uint8_t oldLow = lighting.getMinLightLevel();
754  const std::uint8_t oldHigh = lighting.getMaxLightLevel();
755  lighting.setAmbientLevel(params.lightLevel(), params.lightLevel());
756 
757  for (int px = 0; px < pw; ++px) {
758  for (int py = 0; py < ph; ++py) {
759  BL_LOG_DEBUG << "Rendering patch " << (px * ph + py);
760 
761  const sf::View view(
762  sf::Vector2f{sf::Vector2i{px * PatchSize * core::Properties::PixelsPerTile() +
763  PatchSize * core::Properties::PixelsPerTile() / 2,
764  py * PatchSize * core::Properties::PixelsPerTile() +
765  PatchSize * core::Properties::PixelsPerTile() / 2}},
766  sf::Vector2f{sf::Vector2i{PatchSize * core::Properties::PixelsPerTile(),
767  PatchSize * core::Properties::PixelsPerTile()}});
768  patch.setView(view);
769  patch.clear(sf::Color::Black);
770 
771  refreshRenderRange(view);
772 
773  const auto renderRow = [&patch, this](const core::map::TileLayer& layer, int row) {
774  for (int x = renderRange.left; x < renderRange.left + renderRange.width; ++x) {
775  layer.get(x, row).render(patch, 0.f);

780  for (int x = renderRange.left; x < renderRange.left + renderRange.width; ++x) {
781  core::map::Tile* t = layer(x, row);
782  if (t) t->render(patch, 0.f);
783  }
784  };
785 
786  for (unsigned int i = 0; i < levels.size(); ++i) {
787  const core::map::LayerSet& level = levels[i];
788

791  renderRow(layer, y);

796  renderSorted(layer, y);
