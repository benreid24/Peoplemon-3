330 bool Map::movePossible(const bl::tmap::Position& pos, bl::tmap::Direction dir) const {
331  bl::tmap::Position npos = pos.move(dir);
332  if (npos.position == pos.position) return true;
333  if (!contains(npos)) return false;
334 
335  switch (levels.at(npos.level).collisionLayer().get(npos.position.x, npos.position.y)) {
336  case Collision::Blocked:
337  return false;
338  case Collision::Open:
339  return true;
340  case Collision::TopOpen:
341  return dir == bl::tmap::Direction::Down;
342  case Collision::RightOpen:
343  return dir == bl::tmap::Direction::Left;
344  case Collision::BottomOpen:
345  return dir == bl::tmap::Direction::Up;
346  case Collision::LeftOpen:
347  return dir == bl::tmap::Direction::Right;
348  case Collision::TopRightOpen:
349  return dir == bl::tmap::Direction::Down || dir == bl::tmap::Direction::Left;
350  case Collision::BottomRightOpen:
351  return dir == bl::tmap::Direction::Up || dir == bl::tmap::Direction::Left;
352  case Collision::BottomLeftOpen:
353  return dir == bl::tmap::Direction::Up || dir == bl::tmap::Direction::Right;
354  case Collision::TopLeftOpen:
355  return dir == bl::tmap::Direction::Down || dir == bl::tmap::Direction::Right;
356  case Collision::TopBottomOpen:
357  return dir == bl::tmap::Direction::Up || dir == bl::tmap::Direction::Down;
358  case Collision::LeftRightOpen:
359  return dir == bl::tmap::Direction::Right || dir == bl::tmap::Direction::Left;
360  case Collision::TopClosed:
361  return dir != bl::tmap::Direction::Down;
362  case Collision::RightClosed:
363  return dir != bl::tmap::Direction::Left;
364  case Collision::BottomClosed:
365  case Collision::LedgeHop:
366  return dir != bl::tmap::Direction::Up;
367  case Collision::LeftClosed:
368  return dir != bl::tmap::Direction::Right;
369  case Collision::SurfRequired:
370  return systems->player().state().bag.hasItem(item::Id::JesusShoes);
371  case Collision::WaterfallRequired:
372  return systems->player().state().bag.hasItem(item::Id::JesusShoesUpgrade);
373  default:
374  BL_LOG_WARN << "Bad collision at (" << npos.position.x << ", " << npos.position.y << ")";
375  return false;
376  }

392  script::LegacyWarn::warn(event.script);

394  << event.position.y << ")";

401  const auto visitor = [&movedEvent, &trigger](const Event* ep) {
402  const Event& e = *ep;
403  const sf::IntRect area(e.position, e.areaSize);
404  const bool wasIn = area.contains(
405  {movedEvent.previousPosition.position.x, movedEvent.previousPosition.position.y});
406  const bool isIn =
407  area.contains({movedEvent.position.position.x, movedEvent.position.position.y});
408 
409  switch (e.trigger) {
410  case Event::Trigger::OnEnter:
411  if (!wasIn && isIn) trigger(e);
412  break;
413 
414  case Event::Trigger::OnExit:
415  if (wasIn && !isIn) trigger(e);
416  break;
417 
418  case Event::Trigger::onEnterOrExit:
419  if (wasIn != isIn) trigger(e);
420  break;
421 
422  case Event::Trigger::WhileIn:
423  if (isIn) trigger(e);
424  break;
425 
426  default:
427  break;
428  }
429  };
430  eventRegions.forAllInCellAndNeighbors(
431  movedEvent.position.getWorldPosition(Properties::PixelsPerTile()), visitor);
432 
433  Town* newTown = getTown(movedEvent.position.position);
434  if (newTown != currentTown) {
435  currentTown = newTown;
436  enterTown(currentTown);
437  }

636 void Map::setupTile(unsigned int level, unsigned int layer, const sf::Vector2u& pos) {
637  Tile& tile = levels[level].getLayer(layer).getRef(pos.x, pos.y);
638  if (tile.id() == Tile::Blank) { return; }
639  const glm::vec2 offset(pos.x * Properties::PixelsPerTile(),
640  pos.y * Properties::PixelsPerTile());
641 
642  auto it = renderLevels.begin();
643  std::advance(it, level);
644  auto& zone = it->getZone(levels[level], layer);
645  const bool isYsort = &zone == &it->zones[RenderLevel::Ysort];
646 
647  const auto getBottomDepth = [this, &pos, layer, level, &tile]() {
648  const unsigned int th = tileset->tileHeight(tile.id(), tile.isAnimation());
649  unsigned int size = th / Properties::PixelsPerTile();
650  if (th % Properties::PixelsPerTile() != 0) { ++size; }
651  return getDepthForPosition(level, pos.y + size, layer);
652  };
653 
654  if (tile.isAnimation()) {
655  const auto ait = tileset->sharedAnimations.find(tile.id());
656  bl::ecs::Entity playerEntity = bl::ecs::InvalidEntity;
657  bl::gfx::DiscreteAnimation2DPlayer player;
658  bool isBatchSlideshow = true;
659  if (ait != tileset->sharedAnimations.end()) { playerEntity = ait->second.entity(); }
660  else {
661  auto anim = tileset->getAnim(tile.id());
662  if (!anim) {
663  BL_LOG_ERROR << "Failed to find animation for tile id: " << tile.id();
664  tile.set(Tile::Blank, false);
665  return;
666  }
667  if (anim->isSlideshow()) {
668  player.create(
669  systems->engine(), anim, bl::gfx::DiscreteAnimation2DPlayer::Slideshow);
670  // will be cleaned up when dependency is removed in ECS
671  playerEntity = player.entity();
672  }
673  else {
674  isBatchSlideshow = false;
675  bl::gfx::Animation2D& vanim =
676  *tile.renderObject.emplace<std::shared_ptr<bl::gfx::Animation2D>>(
677  std::make_shared<bl::gfx::Animation2D>(systems->engine(), anim));
678  vanim.addToScene(scene, bl::rc::UpdateSpeed::Static);
679  vanim.getTransform().setPosition(offset);
680  if (isYsort) {
681  const float topDepth = getDepthForPosition(level, pos.y, layer);
682  const float bottomDepth = getBottomDepth();
683  vanim.getTransform().setDepth((topDepth + bottomDepth) * 0.5f);
684  }
685  else { vanim.getTransform().setDepth(getDepthForPosition(level, pos.y, layer)); }
686  }
687  }
688  if (isBatchSlideshow) {
689  bl::gfx::BatchSlideshowSimple& anim =
690  tile.renderObject.emplace<bl::gfx::BatchSlideshowSimple>(
691  systems->engine(), zone.tileAnims, playerEntity);
692 
693  for (auto& v : anim.getVertices()) {
694  v.pos.x += offset.x;
695  v.pos.y += offset.y;
696  if (!isYsort) { v.pos.z = getDepthForPosition(level, pos.y, layer); }
697  }
698  if (isYsort) {
699  const float topDepth = getDepthForPosition(level, pos.y, layer);
700  const float bottomDepth = getBottomDepth();
701  anim.getVertices()[0].pos.z = topDepth;
702  anim.getVertices()[1].pos.z = topDepth;
703  anim.getVertices()[2].pos.z = bottomDepth;
704  anim.getVertices()[3].pos.z = bottomDepth;
705  }
706  anim.commit();
707  }
708  }
709  else {
710  const sf::FloatRect src = tileset->getTileTextureBounds(tile.id());
711  if (src.width < 0.f) {
712  BL_LOG_ERROR << "Failed to find texture for tile id: " << tile.id();
713  return;
714  }
715  bl::gfx::BatchSpriteSimple& sprite =
716  tile.renderObject.emplace<bl::gfx::BatchSpriteSimple>(zone.tileSprites, src);
717  float depth = getDepthForPosition(level, pos.y, layer);
718  if (isYsort) {
719  const float bottomDepth = getBottomDepth();
720  depth = (depth + bottomDepth) * 0.5f;
721  }
722  for (auto& v : sprite.getVertices()) {
723  v.pos.x += offset.x;
724  v.pos.y += offset.y;
725  v.pos.z = depth;
726  }
727  sprite.commit();
728  }
729 }

731 float Map::getDepthForPosition(unsigned int level, unsigned int y, int layer) const {
732  if (level >= levels.size()) {
733  BL_LOG_ERROR << "Got invalid level: " << level;
734  return 0.f;
735  }
736 
737  const LayerSet& lvl = levels[level];
738  const unsigned int maxLayers = std::accumulate(
739  levels.begin(),
740  levels.end(),
741  levels.front().layerCount(),
742  [](unsigned int val, const LayerSet& level) { return std::max(val, level.layerCount()); });
743 
744  // render entities as top ysort layer
745  if (layer < 0) { layer = lvl.bottomLayers().size() + lvl.ysortLayers().size() - 1; }
746 
747  const float zoneBias = y * maxLayers + layer * size.y;
748  const float maxZoneBias = size.y * maxLayers + (maxLayers - 1) * size.y;
749  const float depthPerLevel = maxZoneBias * 3.f;
750  const float levelBias = static_cast<float>(level) * depthPerLevel;
751 
752  // bottom tiles
753  if (layer < lvl.bottomLayers().size()) { return -(zoneBias + levelBias); }
754 
755  // ysort tiles + entities
756  if (layer < lvl.bottomLayers().size() + lvl.ysortLayers().size()) {
757  const float ysortZoneBias = y * maxLayers + layer;
758  return -(ysortZoneBias + maxZoneBias + levelBias);
759  }
760 
761  // top tiles
762  return -(zoneBias + maxZoneBias * 2.f + levelBias);
763 }
