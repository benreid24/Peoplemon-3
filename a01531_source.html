416 void Map::onMapClick(const sf::Vector2f&, const sf::Vector2i& tiles) {
417  BL_LOG_INFO << "Clicked (" << tiles.x << ", " << tiles.y << ")";
418 
419  switch (activeTool) {
420  case Tool::MapEdit:
421  switch (activeSubtool) {
422  case Subtool::Set:
423  switch (tileset.getActiveTool()) {
424  case Tileset::Tiles:
425  if (selectionState == SelectionMade) {
426  mapArea.editMap().setTileArea(levelSelect->getSelectedOption(),
427  layerSelect->getSelectedOption(),
428  selection,
429  tileset.getActiveTile(),
430  false);
431  }
432  else {
433  mapArea.editMap().setTile(levelSelect->getSelectedOption(),
434  layerSelect->getSelectedOption(),
435  tiles,
436  tileset.getActiveTile(),
437  false);
438  }
439  break;
440  case Tileset::Animations:
441  if (selectionState == SelectionMade) {
442  mapArea.editMap().setTileArea(levelSelect->getSelectedOption(),
443  layerSelect->getSelectedOption(),
444  selection,
445  tileset.getActiveAnim(),
446  true);
447  }
448  else {
449  mapArea.editMap().setTile(levelSelect->getSelectedOption(),
450  layerSelect->getSelectedOption(),
451  tiles,
452  tileset.getActiveAnim(),
453  true);
454  }
455  break;
456  case Tileset::CollisionTiles:
457  if (selectionState == SelectionMade) {
458  mapArea.editMap().setCollisionArea(
459  levelSelect->getSelectedOption(), selection, tileset.getActiveCollision());
460  }
461  else {
462  mapArea.editMap().setCollision(
463  levelSelect->getSelectedOption(), tiles, tileset.getActiveCollision());
464  }
465  break;
466  case Tileset::CatchTiles:
467  if (selectionState == SelectionMade) {
468  mapArea.editMap().setCatchArea(
469  levelSelect->getSelectedOption(), selection, tileset.getActiveCatch());
470  }
471  else {
472  mapArea.editMap().setCatch(
473  levelSelect->getSelectedOption(), tiles, tileset.getActiveCatch());
474  }
475  break;
476  default:
477  break;
478  }
479  break;
480 
481  case Subtool::Clear:
482  switch (tileset.getActiveTool()) {
483  case Tileset::Tiles:
484  case Tileset::Animations:
485  if (selectionState == SelectionMade) {
486  mapArea.editMap().setTileArea(levelSelect->getSelectedOption(),
487  layerSelect->getSelectedOption(),
488  selection,
489  core::map::Tile::Blank,
490  false);
491  }
492  else {
493  mapArea.editMap().setTile(levelSelect->getSelectedOption(),
494  layerSelect->getSelectedOption(),
495  tiles,
496  core::map::Tile::Blank,
497  false);
498  }
499  break;
500  case Tileset::CollisionTiles:
501  if (selectionState == SelectionMade) {
502  mapArea.editMap().setCollisionArea(
503  levelSelect->getSelectedOption(), selection, core::map::Collision::Open);
504  }
505  else {
506  mapArea.editMap().setCollision(
507  levelSelect->getSelectedOption(), tiles, core::map::Collision::Open);
508  }
509  break;
510  case Tileset::CatchTiles:
511  if (selectionState == SelectionMade) {
512  mapArea.editMap().setCatchArea(
513  levelSelect->getSelectedOption(), selection, core::map::Catch::NoEncounter);
514  }
515  else {
516  mapArea.editMap().setCatch(
517  levelSelect->getSelectedOption(), tiles, core::map::Catch::NoEncounter);
518  }
519  break;
520  default:
521  break;
522  }
523  break;
524 
525  case Subtool::Select:
526  switch (selectionState) {
527  case SelectionMade:
528  case NoSelection:
529  selectionState = Selecting;
530  selection.left = tiles.x;
531  selection.top = tiles.y;
532  break;
533  case Selecting:
534  selectionState = SelectionMade;
535  {
536  const int minX = std::min(selection.left, tiles.x);
537  const int minY = std::min(selection.top, tiles.y);
538  const int maxX = std::max(selection.left, tiles.x);
539  const int maxY = std::max(selection.top, tiles.y);
540  selection = {minX, minY, maxX - minX + 1, maxY - minY + 1};
541  }
542  break;
543  default:
544  selectionState = NoSelection;
545  break;
546  }
547  break;
548 
549  default:
550  break;
551  }
552 
553  case Tool::Items:
554  case Tool::Lights:
555  case Tool::NPCs:
556  case Tool::Peoplemon:
557  case Tool::Scripts:
558  case Tool::Spawns:
559  case Tool::Metadata:
560  default:
561  break;
562  }
