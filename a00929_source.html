16 
17  if (level < 7) return {};
18 
19  Stats stats;

21  const unsigned int iterCount = l * l / 12;
22  const std::vector<Id>& ids = Peoplemon::validIds();
23  for (unsigned int i = 0; i < iterCount; ++i) {
