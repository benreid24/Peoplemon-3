485 void PeoplemonMenu::chosen() {
486  bool closeMenu = true;
487 
488  switch (context) {
489  case Context::BattleFaint:
490  case Context::BattleSwitch:
491  case Context::BattleMustSwitch:
492  if (systems.player().state().peoplemon[mover1].currentHp() == 0) {
493  resetAction();
494  state = MenuState::ShowingMessage;
495  systems.hud().displayMessage("Unconscious Peoplemon can't fight!",
496  std::bind(&PeoplemonMenu::messageDone, this));
497  return;
498  }
499  break;
500 
501  case Context::BattleReviveSwitch:
502  if (systems.player().state().peoplemon[mover1].currentHp() != 0) {
503  resetAction();
504  state = MenuState::ShowingMessage;
505  systems.hud().displayMessage(
506  "Choose a dead Peoplemon so the sacrifice was not in vain!",
507  std::bind(&PeoplemonMenu::messageDone, this));
508  return;
509  }
510  break;
511 
512  case Context::GiveItem:
513  if (systems.player().state().peoplemon[mover1].holdItem() != core::item::Id::None) {
514  resetAction();
515  state = MenuState::ShowingMessage;
516  systems.hud().displayMessage(systems.player().state().peoplemon[mover1].name() +
517  " is already holding something",
518  std::bind(&PeoplemonMenu::messageDone, this));
519  return;
520  }
521  break;
522 
523  case Context::UseItem:
524  closeMenu = false;
525  actionOpen = false;
526 
527  if (core::item::Item::hasEffectOnPeoplemon(useItem,
528  systems.player().state().peoplemon[mover1])) {
529  core::pplmn::OwnedPeoplemon& ppl = systems.player().state().peoplemon[mover1];
530  core::item::Item::useOnPeoplemon(useItem, ppl, &systems.player().state().peoplemon);
531  systems.player().state().bag.removeItem(useItem);
532  systems.hud().displayMessage(core::item::Item::getUseLine(useItem, ppl),
533  std::bind(&PeoplemonMenu::messageDone, this));
534  buttons[mover1]->sync(ppl);
535  state = MenuState::UsingItem;
536  }
537  else if (core::item::Item::getCategory(useItem) == core::item::Category::TM) {
538  const core::pplmn::MoveId mid = core::item::Item::getTmMove(useItem);
539  if (mid != core::pplmn::MoveId::Unknown) {
540  core::pplmn::OwnedPeoplemon& ppl = systems.player().state().peoplemon[mover1];
541  const std::string move = core::pplmn::Move::name(mid);
542 
543  int i = -1;
544  for (int j = 0; j < 4; ++j) {
545  if (ppl.knownMoves()[j].id == core::pplmn::MoveId::Unknown) {
546  i = j;
547  break;
548  }
549  }
550 
551  // already knows 4 moves
552  if (i < 0) {
553  systems.hud().displayMessage(ppl.name() + " is trying to learn " + move +
554  " but already knows 4 moves.");
555  systems.hud().promptUser(
556  "Should " + ppl.name() + " forget a move to learn " + move + "?",
557  {"Yes", "No"},
558  std::bind(&PeoplemonMenu::confirmMoveDelete, this, std::placeholders::_1));
559  state = MenuState::WaitingForgetConfirm;
560  }
561  else {
562  systems.player().state().bag.removeItem(useItem);
563  ppl.knownMoves()[i] = core::pplmn::OwnedMove(mid);
564  systems.hud().displayMessage(ppl.name() + " learned " + move + "!",
565  std::bind(&PeoplemonMenu::messageDone, this));
566  state = MenuState::UsingItemWaitMessage;
567  }
568  }
569  else {
570  BL_LOG_ERROR << "Invalid tm: " << useItem;
571  systems.hud().displayMessage("Sorry but this TM is bunk.",
572  std::bind(&PeoplemonMenu::messageDone, this));
573  state = MenuState::ShowingMessage;
574  }
575  }
576  else {
577  resetAction();
578  state = MenuState::ShowingMessage;
579  systems.hud().displayMessage("It will not have any effect.",
580  std::bind(&PeoplemonMenu::messageDone, this));
581  }
582 
583  break;
584 
585  case Context::UseItemBattle:
586  // no extra logic needed. if the item has no effect we just let them waste their turn
587  [[fallthrough]];
588 
589  default:
590  break;
591  }
592 
593  inputDriver.drive(nullptr);
594 
595  if (chosenPeoplemon) *chosenPeoplemon = mover1;
596  if (closeMenu) systems.engine().popState();
597 }
