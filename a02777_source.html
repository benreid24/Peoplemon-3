14 void shiftDownJumps(std::vector<Conversation::Node>& nodes, unsigned int i) {
15  const auto updateJump = [&nodes, i](std::uint32_t& jump) {
16  const auto isSimpleNext = [&nodes](unsigned int i) {
17  if (i >= nodes.size()) return false;
18  switch (nodes[i].getType()) {
19  case Conversation::Node::Talk:
20  case Conversation::Node::GiveMoney:
21  case Conversation::Node::GiveItem:
22  case Conversation::Node::RunScript:
23  case Conversation::Node::SetSaveFlag:
24  return true;
25 
26  default:
27  return false;
28  }
29  };
30 
31  if (jump == i) {
32  if (isSimpleNext(jump)) {
33  const std::uint32_t j = nodes[jump].next();
34  jump = j > i ? j - 1 : j;
35  }
36  else {
37  jump = EndNode;
38  }
39  }
40  else if (jump > i)
41  --jump;
42  };
43 
44  for (unsigned int j = 0; j < nodes.size(); ++j) {
45  Conversation::Node& node = nodes[j];
46 
47  // Invalidate or shift down
48  updateJump(node.nextOnPass());
49  updateJump(node.nextOnReject());
50 
51  // Choices
52  if (node.getType() == Conversation::Node::Prompt) {
53  for (auto& pair : node.choices()) {
54  // Invalidate or shift down
55  updateJump(pair.second);
56  }
57  }
58  }
59 }

209  else {

211  }

223  const auto connectJump = [&nodes](unsigned int orig, unsigned int next) {
224  for (Conversation::Node& node : nodes) {
225  if (node.nextOnPass() == orig)

241  }
242  }

247  const unsigned int next =
248  nodes[i].next() > i ? nodes[i].next() - 1 : nodes[i].next();
