542 bool Map::movePossible(const component::Position& pos, component::Direction dir) const {
543  component::Position npos = pos.move(dir);
544  if (npos.positionTiles() == pos.positionTiles()) return true;
545  if (!contains(npos)) return false;
546 
547  switch (levels.at(npos.level)
548  .collisionLayer()
549  .get(npos.positionTiles().x, npos.positionTiles().y)) {
550  case Collision::Blocked:
551  return false;
552  case Collision::Open:
553  return true;
554  case Collision::TopOpen:
555  return dir == component::Direction::Down;
556  case Collision::RightOpen:
557  return dir == component::Direction::Left;
558  case Collision::BottomOpen:
559  return dir == component::Direction::Up;
560  case Collision::LeftOpen:
561  return dir == component::Direction::Right;
562  case Collision::TopRightOpen:
563  return dir == component::Direction::Down || dir == component::Direction::Left;
564  case Collision::BottomRightOpen:
565  return dir == component::Direction::Up || dir == component::Direction::Left;
566  case Collision::BottomLeftOpen:
567  return dir == component::Direction::Up || dir == component::Direction::Right;
568  case Collision::TopLeftOpen:
569  return dir == component::Direction::Down || dir == component::Direction::Right;
570  case Collision::TopBottomOpen:
571  return dir == component::Direction::Up || dir == component::Direction::Down;
572  case Collision::LeftRightOpen:
573  return dir == component::Direction::Right || dir == component::Direction::Left;
574  case Collision::TopClosed:
575  return dir != component::Direction::Down;
576  case Collision::RightClosed:
577  return dir != component::Direction::Left;
578  case Collision::BottomClosed:
579  return dir != component::Direction::Up;
580  case Collision::LeftClosed:
581  return dir != component::Direction::Right;
582  case Collision::SurfRequired:

584  case Collision::WaterfallRequired:

586  default:

617  case Event::Trigger::OnEnter:
618  if (!wasIn && isIn) trigger(e);
619  break;
620 
621  case Event::Trigger::OnExit:
622  if (wasIn && !isIn) trigger(e);
623  break;
624 
625  case Event::Trigger::onEnterOrExit:
626  if (wasIn != isIn) trigger(e);
627  break;
628 
629  case Event::Trigger::WhileIn:
630  if (isIn) trigger(e);
631  break;
632 
633  default:
634  break;
635  }
636  }
637

639  if (newTown != currentTown) {
640  currentTown = newTown;
641  enterTown(currentTown);
642  }

661  const auto trigger = [this, interactor, &pos](const Event& event) {
662  script::LegacyWarn::warn(event.script);
663  BL_LOG_INFO << interactor << " triggered event at (" << pos.positionTiles().x << ", "
664  << pos.positionTiles().y << ")";
665  bl::script::Script s(event.script,
666  script::MapEventContext(*systems, interactor, event, pos));
667  s.run(&systems->engine().scriptManager());
668  };
669 
670  const auto range =
671  eventRegions.getCellAndNeighbors(pos.positionPixels().x, pos.positionPixels().y);
672  for (const auto& it : range) {
673  const Event& e = *it.get();
674  const sf::IntRect area(e.position, e.areaSize);
675  if (area.contains(pos.positionTiles()) && e.trigger == Event::Trigger::OnInteract) {
676  trigger(e);
677  return true;
678  }
679  }
680 
681  return false;

699 void Map::refreshRenderRange(const sf::View& view) const {
700  static const sf::Vector2i ExtraRender =
701  sf::Vector2i(Properties::ExtraRenderTiles(), Properties::ExtraRenderTiles());
702 
703  const sf::Vector2f cornerPixels = view.getCenter() - view.getSize() * 0.5f;
704  sf::Vector2i corner =
705  static_cast<sf::Vector2i>(cornerPixels) / Properties::PixelsPerTile() - ExtraRender;
706  if (corner.x < 0) corner.x = 0;
707  if (corner.y < 0) corner.y = 0;
708 
709  sf::Vector2i wsize =
710  static_cast<sf::Vector2i>(view.getSize()) / Properties::PixelsPerTile() + ExtraRender * 2;
711  if (corner.x + wsize.x > size.x) wsize.x = size.x - corner.x;
712  if (corner.y + wsize.y > size.y) wsize.y = size.y - corner.y;
713  renderRange = {corner, wsize};
714 }
