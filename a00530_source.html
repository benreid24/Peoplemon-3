48  using Animation = cmd::Animation;
49 
50  switch (cmd.getType()) {
51  case Command::Type::DisplayMessage:
52  printer.setMessage(battleState, cmd.getMessage());
53  break;
54 
55  case Command::Type::PlayAnimation: {
56  const bool userIsPlayer =
57  cmd.getAnimation().forHostBattler() == battleState.localPlayer().isHost();
58 
59  switch (cmd.getAnimation().getType()) {
60  case Animation::Type::UseMove:
61  moveAnimation.playAnimation(userIsPlayer ? view::MoveAnimation::Player :
62  view::MoveAnimation::Opponent,
63  cmd.getAnimation().getMove());
64  break;
65 
66  case Animation::Type::PlayerFirstSendout:
67  localPeoplemon.triggerAnimation(Animation::Type::SendOut);
68  break;
69 
70  case Animation::Type::OpponentFirstSendout:
71  opponentPeoplemon.triggerAnimation(Animation::Type::SendOut);
72  break;
73 
74  default: {
75  auto& anim = userIsPlayer ? localPeoplemon : opponentPeoplemon;
76  anim.triggerAnimation(cmd.getAnimation());
77  } break;
78  }
79  } break;
80 
81  case Command::Type::SyncStateNoSwitch:
82  statBoxes.sync();
83  break;
84 
85  case Command::Type::SyncStateSwitch: {
86  const bool isPlayer = cmd.forHost() == battleState.localPlayer().isHost();
87 
88  if (isPlayer) {
89  statBoxes.setPlayer(&battleState.localPlayer().activePeoplemon());
90  playerMenu.setPeoplemon(battleState.localPlayer().outNowIndex(),
91  battleState.localPlayer().activePeoplemon());
92  localPeoplemon.setPeoplemon(battleState.localPlayer().activePeoplemon().base().id());
93  }
94  else {
95  statBoxes.setOpponent(&battleState.enemy().activePeoplemon());
96  opponentPeoplemon.setPeoplemon(battleState.enemy().activePeoplemon().base().id());
97  }
98  moveAnimation.ensureLoaded(battleState.localPlayer().activePeoplemon(),
99  battleState.enemy().activePeoplemon());
100  } break;
101 
102  default:
103  BL_LOG_ERROR << "Unknown command type: " << cmd.getType();
104  break;
105  }
