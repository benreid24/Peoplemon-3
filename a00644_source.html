347 void TrainerEditorWindow::load(const std::string& file) {
348  core::file::Trainer trainer;
349  if (trainer.load(FileUtil::joinPath(core::Properties::TrainerPath(), file))) {
350  nameEntry->setInput(trainer.name);
351  animLabel->setText(trainer.animation);
352  bbLabel->setText(trainer.prebattleConversation);
353  abLabel->setText(trainer.postBattleConversation);
354  loseLineEntry->setInput(trainer.lostBattleLine);
355  visionRangeEntry->setSelectedOption(trainer.visionRange);
356  behaviorEditor.configure(parent, trainer.behavior);
357  items = trainer.items;
358  for (const auto item : items) { itemSelectBox->addOption(core::item::Item::getName(item)); }
359  peoplemon = trainer.peoplemon;
360  for (const auto& ppl : peoplemon) { pplBox->addOption(pplToStr(ppl)); }
361  }
362  else {
363  bl::dialog::tinyfd_messageBox(
364  "Error", std::string("Failed to load Trainer:\n" + file).c_str(), "ok", "error", 1);
365  }
366 }

368 bool TrainerEditorWindow::validate(bool saving) const {
369  if (saving) {
370  if (fileLabel->getText() == EmptyFile) {
371  bl::dialog::tinyfd_messageBox("Warning", "Please select a file", "ok", "warning", 1);
372  return false;
373  }
374  }
375  else {
376  if (!FileUtil::exists(
377  FileUtil::joinPath(core::Properties::TrainerPath(), fileLabel->getText()))) {
378  bl::dialog::tinyfd_messageBox("Warning", "Bad file selected", "ok", "warning", 1);
379  return false;
380  }
381  }
382  if (nameEntry->getInput().empty()) {
383  bl::dialog::tinyfd_messageBox("Warning", "Enter a name", "ok", "warning", 1);
384  return false;
385  }
386  if (!FileUtil::directoryExists(
387  FileUtil::joinPath(core::Properties::CharacterAnimationPath(), animLabel->getText()))) {
388  bl::dialog::tinyfd_messageBox("Warning", "Bad animation", "ok", "warning", 1);
389  return false;
390  }
391  std::string p = FileUtil::joinPath(core::Properties::ConversationPath(), bbLabel->getText());
392  if (!FileUtil::exists(p)) {
393  BL_LOG_INFO << p;
394  bl::dialog::tinyfd_messageBox(
395  "Warning", "Bad before-battle conversation", "ok", "warning", 1);
396  return false;
397  }
398  p = FileUtil::joinPath(core::Properties::ConversationPath(), abLabel->getText());
399  if (!FileUtil::exists(p)) {
400  BL_LOG_INFO << p;
401  bl::dialog::tinyfd_messageBox(
402  "Warning", "Bad afer-battle conversation", "ok", "warning", 1);
403  return false;
404  }
405  if (loseLineEntry->getInput().empty()) {
406  bl::dialog::tinyfd_messageBox("Warning", "Lose line is empty", "ok", "warning", 1);
407  return false;
408  }
409  if (behaviorEditor.getValue().type() == core::file::Behavior::Type::FollowingPath &&
410  behaviorEditor.getValue().path().paces.empty()) {
411  bl::dialog::tinyfd_messageBox("Warning", "Behavior path is empty", "ok", "warning", 1);
412  return false;
413  }
414  if (peoplemon.empty()) {
415  bl::dialog::tinyfd_messageBox("Warning", "Trainer has no peoplemon", "ok", "warning", 1);
416  return false;
417  }
418  return true;
419 }
