363 bool Map::movePossible(const component::Position& pos, component::Direction dir) const {
364  component::Position npos = pos.move(dir);
365  if (npos.positionTiles() == pos.positionTiles()) return true;
366  if (!contains(npos)) return false;
367 
368  switch (levels.at(npos.level)
369  .collisionLayer()
370  .get(npos.positionTiles().x, npos.positionTiles().y)) {
371  case Collision::Blocked:
372  return false;
373  case Collision::Open:
374  return true;
375  case Collision::TopOpen:
376  return dir == component::Direction::Down;
377  case Collision::RightOpen:
378  return dir == component::Direction::Left;
379  case Collision::BottomOpen:
380  return dir == component::Direction::Up;
381  case Collision::LeftOpen:
382  return dir == component::Direction::Right;
383  case Collision::TopRightOpen:
384  return dir == component::Direction::Down || dir == component::Direction::Left;
385  case Collision::BottomRightOpen:
386  return dir == component::Direction::Up || dir == component::Direction::Left;
387  case Collision::BottomLeftOpen:
388  return dir == component::Direction::Up || dir == component::Direction::Right;
389  case Collision::TopLeftOpen:
390  return dir == component::Direction::Down || dir == component::Direction::Right;
391  case Collision::TopBottomOpen:
392  return dir == component::Direction::Up || dir == component::Direction::Down;
393  case Collision::LeftRightOpen:
394  return dir == component::Direction::Right || dir == component::Direction::Left;
395  case Collision::TopClosed:
396  return dir != component::Direction::Down;
397  case Collision::RightClosed:
398  return dir != component::Direction::Left;
399  case Collision::BottomClosed:
400  case Collision::LedgeHop:
401  return dir != component::Direction::Up;
402  case Collision::LeftClosed:
403  return dir != component::Direction::Right;
404  case Collision::SurfRequired:
405  return systems->player().state().bag.hasItem(item::Id::JesusShoes);
406  case Collision::WaterfallRequired:
407  return systems->player().state().bag.hasItem(item::Id::JesusShoesUpgrade);
408  default:
409  BL_LOG_WARN << "Bad collision at (" << npos.positionTiles().x << ", "
410  << npos.positionTiles().y << ")";

615  if (pos.level >= levels.size() || tiles.x < 0 ||
616  static_cast<unsigned int>(tiles.x) >= levels.front().catchLayer().width() || tiles.y < 0 ||
617  static_cast<unsigned int>(tiles.y) >= levels.front().catchLayer().height()) {
618  BL_LOG_ERROR << "Out of bounds position: " << pos;
619  return nullptr;
620  }
621 
622  std::uint8_t ci = levels[pos.level].catchLayer().get(tiles.x, tiles.y);
623  if (ci == 0) { return nullptr; }
624  --ci; // convert to index
625  if (ci >= catchRegionsField.size()) {
626  BL_LOG_ERROR << "Bad catch zone index (" << static_cast<int>(ci)
627  << ") at position: " << pos;
628  return nullptr;
629  }
630  return &catchRegionsField[ci];
631 }
