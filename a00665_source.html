159 void LightingSystem::update(float dt) {
160  if (weatherModifier != targetWeatherModifier) {
161  weatherResidual += AdjustSpeed * dt;
162  const float p = std::floor(weatherResidual);
163  if (p > 0.f) {
164  weatherResidual -= p;
165  if (targetWeatherModifier > weatherModifier) {
166  weatherModifier += static_cast<int>(p);
167  if (weatherModifier > targetWeatherModifier) {
168  weatherModifier = targetWeatherModifier;
169  }
170  }
171  else {
172  weatherModifier -= static_cast<int>(p);
173  if (weatherModifier < targetWeatherModifier) {
174  weatherModifier = targetWeatherModifier;
175  }
176  }
177  }
178  }
179  else { weatherResidual = 0.f; }
180 }

226 void LightingSystem::observe(const event::TimeChange& now) {
227  if (adjustsForSunlight()) {
228  const float x = now.newTime.hour * 60 + now.newTime.minute;
229  const float n = 0.7;
230  sunlightFactor = 1.f - (0.5 * std::cos(3.1415926 * x / 720) + 0.5) *
231  ((1 - n) * (720 - x) * (720 - x) / 518400 + n);
232  }
233  else { sunlightFactor = 1.f; }
