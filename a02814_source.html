173 
174  for (const auto& tpair : textureFiles) {
175  textures.emplace(

178  .load(bl::util::FileUtil::joinPath(Properties::MapTilePath(), tpair.second))
179  .data);
180  if (tpair.first >= nextTextureId) nextTextureId = tpair.first + 1;
181  }
182

184  auto it = anims

187  .load(bl::util::FileUtil::joinPath(
188  Properties::MapAnimationPath(), apair.second))
189  .data)
190  .first;
191  if (it->second->isLooping()) { sharedAnimations.emplace(apair.first, *it->second); }
192  if (apair.first >= nextAnimationId) nextAnimationId = apair.first + 1;
193  }
194 
195  return true;
