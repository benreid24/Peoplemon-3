41  BL_LOG_INFO << "Entity " << interactor << " interacted with entity: " << interacted;
42

45  return false;

109 void Interaction::processConversationNode() {
110  if (interactingEntity == bl::entity::InvalidEntity) return;
111  if (currentConversation.finished()) {
112  if (interactingTrainer) { startBattle(); }
113  else {
114  owner.controllable().resetEntityLock(interactingEntity);
115  }
116  return;
117  }
118 
119  using E = file::Conversation::Node::Type;
120  const file::Conversation::Node& node = currentConversation.currentNode();
121  switch (node.getType()) {
122  case E::Talk:
123  owner.hud().displayMessage(node.message(), std::bind(&Interaction::continuePressed, this));
124  break;
125 
126  case E::Prompt: {
127  std::vector<std::string> choices;
128  choices.reserve(node.choices().size());
129  for (const auto& pair : node.choices()) choices.push_back(pair.first);
130  owner.hud().promptUser(node.message(),
131  choices,
132  std::bind(&Interaction::choiceMade, this, std::placeholders::_1));
133  } break;
134 
135  case E::GiveItem:
136  owner.player().state().bag.addItem(node.item().id);
137  if (node.item().afterPrompt) {
138  // TODO - add prefix to item metadata for a/the use
139  owner.hud().displayMessage("Got a " + item::Item::getName(node.item().id),
140  std::bind(&Interaction::continuePressed, this));
141  }
142  break;
143 
144  case E::TakeItem:
145  if (node.item().beforePrompt) {
146  owner.hud().promptUser(
147  "Hand over a " + item::Item::getName(node.item().id),
148  {"Yes", "No"},
149  std::bind(&Interaction::giveItemDecided, this, std::placeholders::_1));
150  }
151  else {
152  if (owner.player().state().bag.removeItem(
153  currentConversation.currentNode().item().id)) {
154  currentConversation.notifyCheckPassed();
155  processConversationNode();
156  }
157  }
158  break;
159 
160  case E::GiveMoney:
161  owner.player().state().monei += node.money();
162  owner.hud().displayMessage("Received " + std::to_string(node.money()) + " monies",
163  std::bind(&Interaction::continuePressed, this));
164  break;
165 
166  case E::TakeMoney:
167  owner.hud().promptUser(
168  "Fork over " + std::to_string(node.money()) + " monies?",
169  {"Yes", "No"},
170  std::bind(&Interaction::giveMoneyDecided, this, std::placeholders::_1));
171  break;
172 
173  default:
174  BL_LOG_ERROR << "Invalid conversation node type " << node.getType() << ", terminating";
175  owner.controllable().resetEntityLock(interactingEntity);
176  interactingEntity = bl::entity::InvalidEntity;
177  break;
178  }
