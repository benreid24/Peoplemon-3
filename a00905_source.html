340 void BagMenu::useItem() {
341  if (context == Context::BattleUse) {
342  if (!core::item::Item::canUseInBattle(selectedItem->getItem().id)) {
343  state = MenuState::ShowingMessage;
344  systems.hud().displayMessage("This item cannot be used in battle!",
345  std::bind(&BagMenu::messageDone, this));
346  return;
347  }
348  }
349 
350  const core::item::Type type = core::item::Item::getType(selectedItem->getItem().id);
351  if (needPickPeoplemon(selectedItem->getItem().id, type)) {
352  state = MenuState::ChoosingItemPeoplemon;
353  systems.engine().pushState(PeoplemonMenu::create(systems,
354  context == Context::BattleUse ?
355  PeoplemonMenu::Context::UseItemBattle :
356  PeoplemonMenu::Context::UseItem,
357  outNow,
358  itemPeoplemon,
359  selectedItem->getItem().id));
360  }
361  else {
362  if (context == Context::BattleUse) {
363  if (result) *result = selectedItem->getItem().id;
364  systems.engine().popState();
365  }
366  else {
367  switch (type) {
368  case core::item::Type::KeyItem:
369  // TODO - use key items
370  break;
371  case core::item::Type::PlayerModifier:
372  if (core::item::Item::hasEffectOnPlayer(selectedItem->getItem().id,
373  systems.player().state())) {
374  core::item::Item::useOnPlayer(selectedItem->getItem().id,
375  systems.player().state());
376  systems.player().state().bag.removeItem(selectedItem->getItem().id);
377  systems.hud().displayMessage(
378  core::item::Item::getUseLine(selectedItem->getItem().id),
379  std::bind(&BagMenu::messageDone, this));
380  }
381  else {
382  systems.hud().displayMessage("It won't have any effect",
383  std::bind(&BagMenu::messageDone, this));
384  }
385  state = MenuState::ShowingMessage;
386  break;
387  case core::item::Type::Useless:
388  systems.hud().displayMessage(
389  "The Professor's voice echos in your head: \"That thing is completely useless! "
390  "What are you expecting to happen bro?\"",
391  std::bind(&BagMenu::messageDone, this));
392  state = MenuState::ShowingMessage;
393  break;
394  default:
395  systems.hud().displayMessage(
396  "The Professor's voice echos in your head: \"You can't use that here bro\"",
397  std::bind(&BagMenu::messageDone, this));
398  state = MenuState::ShowingMessage;
399  break;
400  }
401  }
