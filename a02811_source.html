479  levelSelect->getSelectedOption());
480  break;
481 
482  case Tileset::CatchTiles:
483  mapArea.editMap().setRenderOverlay(component::EditMap::RenderOverlay::CatchTiles,
484  levelSelect->getSelectedOption());
485  break;
486 
487  case Tileset::TownTiles:
488  mapArea.editMap().setRenderOverlay(component::EditMap::RenderOverlay::Towns, 0);
489  break;
490 
491  default:
492  switch (activeTool) {
493  case Tool::Events:
494  mapArea.editMap().setRenderOverlay(component::EditMap::RenderOverlay::Events, 0);
495  break;
496 
497  case Tool::Spawns:
498  mapArea.editMap().setRenderOverlay(component::EditMap::RenderOverlay::Spawns,
499  levelSelect->getSelectedOption());
500  break;
501 
502  default:
503  mapArea.editMap().setRenderOverlay(component::EditMap::RenderOverlay::None, 0);
504  break;
505  }
506  break;
507  }
508 
509  if (mapArea.editMap().unsavedChanges() || tileset.unsavedChanges()) {
510  saveMapBut->setColor(sf::Color(200, 185, 20), sf::Color::Red);
