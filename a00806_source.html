100 void Flight::update(float dt) {
101  switch (state) {
102  case State::Rising:
103  riseState.height = std::min(riseState.height + RiseRate * dt, RiseHeight);
104  playerPos->setPixels({playerPos->positionPixels().x, riseState.startY - riseState.height});
105  playerAnim->updateShadow(riseState.height,
106  ShadowSize - (riseState.height / RiseHeight) * ShadowShrinkage);
107  if (riseState.height >= RiseHeight) {
108  state = State::Rotating;
109  flightDest = destination.positionPixels();
110  flightDest.y -= RiseHeight;
111  unitVelocity = flightDest - playerPos->positionPixels();
112  const float m =
113  std::sqrt(unitVelocity.x * unitVelocity.x + unitVelocity.y * unitVelocity.y);
114  unitVelocity.x /= m;
115  unitVelocity.y /= m;
116  rotateState.angle = 0.f;
117  rotateState.targetAngle = bl::math::radiansToDegrees(std::atan2(
118  flightDest.y - playerPos->positionPixels().y,
119  flightDest.x - playerPos->positionPixels().x)) +
120  90.f;
121  rotateState.rotateRate = (rotateState.targetAngle - rotateState.angle) / RotateTime;
122  }
123  break;
124 
125  case State::Rotating:
126  case State::UnRotating:
127  rotateState.angle += rotateState.rotateRate * dt;
128  playerAnim->setAngle(rotateState.angle);
129  if (std::abs(rotateState.angle - rotateState.targetAngle) < 1.5f) {
130  rotateState.angle = rotateState.targetAngle;
131  playerAnim->setAngle(rotateState.angle);
132 
133  if (state == State::Rotating) {
134  const float pa = rotateState.targetAngle;
135  state = State::Accelerating;
136  flyState.angle = pa; // probably not required but eh
137  flyState.velocity = 0.f;
138  flyState.ogDistSqrd = distSqrd(*playerPos, flightDest);
139  flyState.maxSpeed = std::min(std::sqrt(flyState.ogDistSqrd) / MinFlyTime, MaxSpeed);
140  }
141  else {
142  state = State::Descending;
143  riseState.height = RiseHeight;
144  const sf::Vector2f pixels = playerPos->positionPixels();
145  playerPos->setTiles(destination.positionTiles());
146  playerPos->setPixels(pixels);
147  }
148  }
149  break;
150 
151  case State::Accelerating: {
152  movePlayer(dt);
153  const float cDist = distSqrd(*playerPos, flightDest);
154  const float percent = 1.f - (cDist / flyState.ogDistSqrd) + 0.005f;
155  flyState.velocity =
156  percent <= 0.125f ? percent * 8.f * flyState.maxSpeed : flyState.maxSpeed;
157  if (flyState.velocity >= flyState.maxSpeed) {
158  flyState.velocity = flyState.maxSpeed;
159  state = State::Flying;
160  }
161  if (cameraShake) {
162  cameraShake->setMagnitude(flyState.velocity / flyState.maxSpeed * ShakeMagnitude);
163  cameraShake->setShakesPerSecond(flyState.velocity / flyState.maxSpeed * ShakesPerSec);
164  }
165  } break;
166 
167  case State::Flying: {
168  movePlayer(dt);
169  const float cDist = distSqrd(*playerPos, flightDest);
170  const float percent = 1.f - cDist / flyState.ogDistSqrd;
171  if (percent >= 0.98f) { state = State::Deccelerating; }
172  } break;
173 
174  case State::Deccelerating: {
175  movePlayer(dt);
176  const float cDist = distSqrd(*playerPos, flightDest);
177  const float percent = 1.f - cDist / flyState.ogDistSqrd;
178  flyState.velocity = std::max((1.f - percent) * 50.f * flyState.maxSpeed, 32.f * 6.f);
179  if (cDist <= 3.4f || flyState.velocity == 0.f) {
180  if (cameraShake) { cameraShake->setMagnitude(0.f); }
181  playerPos->setPixels(flightDest);
182  const float pa = flyState.angle;
183  state = State::UnRotating;
184  rotateState.angle = pa;
185  rotateState.targetAngle = 0.f;
186  rotateState.rotateRate = (rotateState.targetAngle - rotateState.angle) / RotateTime;
187  }
188  else if (cameraShake) {
189  cameraShake->setMagnitude(flyState.velocity / flyState.maxSpeed * ShakeMagnitude);
190  cameraShake->setShakesPerSecond(flyState.velocity / flyState.maxSpeed * ShakesPerSec);
191  }
192  } break;
193 
194  case State::Descending:
195  riseState.height = std::max(riseState.height - RiseRate * dt, 0.f);
196  playerPos->setPixels(
197  {playerPos->positionPixels().x, destination.positionPixels().y - riseState.height});
198  playerAnim->updateShadow(riseState.height,
199  ShadowSize - (riseState.height / RiseHeight) * ShadowShrinkage);
200  if (riseState.height == 0.f) {
201  playerAnim->removeShadow();

203  *playerPos = destination;
204  playerPos->direction = component::Direction::Down;
205  state = State::Idle;
206  auto* cam =
207  owner.engine().renderer().getObserver().getCurrentCamera<bl::cam::Camera2D>();
208  if (cam) { cam->removeAffector(cameraShake); }
209  bl::event::Dispatcher::dispatch<event::EntityMoved>(
210  {owner.player().player(), prev, *playerPos});
