728  if (state->localPlayer().actionSelected()) {
729  if (state->localPlayer().shouldContinue()) { setBattleState(Stage::CheckFaint); }
730  else {
731  setBattleState(Stage::Completed);
732  }
733  }
734  break;
735 
736  case Stage::WaitingFaintSwitch:
737  if (currentFainter->actionSelected()) { setBattleState(Stage::FaintSwitching); }
738  break;
739 
740  case Stage::FaintSwitching:
741  setBattleState(Stage::AfterFaintSwitch);
742  break;
743 
744  case Stage::AfterFaintSwitch:
745  setBattleState(Stage::RoundEnd);
746  break;
747 
748  case Stage::PeoplemonCaught:
749  case Stage::PeoplemonCloned:
750  if (battle->view.playerChoseToSetName()) { setBattleState(Stage::ChoosingNickname); }
751  else {
752  setBattleState(Stage::SavingPeoplemon);
753  }
754  break;
755 
756  case Stage::ChoosingNickname:
757  setBattleState(Stage::SavingPeoplemon);
758  break;
759 
760  case Stage::SavingPeoplemon:
761  setBattleState(state->activeBattler().chosenItem() != item::Id::CloneBall ?
762  Stage::Completed :
763  Stage::NextBattler);
764  break;
765 
766  case Stage::TrainerDefeated:
767  case Stage::NetworkDefeated:
768  case Stage::NetworkLost:
769  case Stage::Whiteout:
770  setBattleState(Stage::Completed);
771  break;
772 
773  case Stage::NetworkDisconnect:
774  case Stage::NextBattler:
775  case Stage::Completed:
776  case Stage::RoundStart:
777  case Stage::CheckPlayerContinue:
778  case Stage::CheckFaint:
779  case Stage::XpAwardBegin:
780  case Stage::RoundEnd:
781  case Stage::BeforeRun:
782  // do nothing, these are intermediate states
783  break;
784 
785  default:
786  BL_LOG_CRITICAL << "Invalid battle stage: " << state->currentStage();
787  setBattleState(Stage::Completed);
788  break;
789  }
790  }

806 BattleState::Stage LocalBattleController::getNextStage(BattleState::Stage ns) {
807  using Stage = BattleState::Stage;
808 
809  switch (ns) {
810  case Stage::NextBattler:
811  if (state->inactiveBattler().activePeoplemon().base().currentHp() == 0 ||
812  state->activeBattler().activePeoplemon().base().currentHp() == 0) {
813  return Stage::Fainting;
814  }
815  else {
816  const BattleState::Stage next = state->nextTurn();
817  queueCommand({Command::SyncStateNoSwitch});
818  return next;
819  }
820 
821  case Stage::RoundStart: {
822  queueCommand({Command::SyncStateNoSwitch});
823  int pp = getPriority(state->localPlayer());
824  int op = getPriority(state->enemy());
825  firstTurn = false;
826 
827  bool pfirst = true;
828  if (pp < op) { pfirst = false; }
829  else if (pp == op && state->localPlayer().chosenAction() == TurnAction::Fight) {
830  const int lps = getSpeed(state->localPlayer());
831  const int eps = getSpeed(state->enemy());
832  if (eps > lps) { pfirst = false; }
833  else if (lps == eps) {
834  pfirst = bl::util::Random::get<int>(0, 100) <= 50;
835  }
836  }
837 
838  state->beginRound(pfirst);
839  queueCommand({Command::SyncStateNoSwitch});
840  state->localPlayer().activePeoplemon().notifyInBattle();
841  state->enemy().activePeoplemon().notifyInBattle();
842  return Stage::TurnStart;
843  }
844 
845  case Stage::RoundEnd:
846  if (state->inactiveBattler().activePeoplemon().base().currentHp() == 0 ||
847  state->activeBattler().activePeoplemon().base().currentHp() == 0) {
848  return Stage::Fainting;
849  }
850  else {
851  queueCommand({Command::SyncStateNoSwitch});
852  return Stage::WaitingChoices;
853  }
854 
855  case Stage::CheckPlayerContinue:
856  if (currentFainter == &state->localPlayer() &&
857  battle->type == Battle::Type::WildPeoplemon) {
858  state->localPlayer().startChooseToContinue();
859  return Stage::WaitingPlayerContinue;
860  }
861  return Stage::CheckFaint;
862 
863  case Stage::CheckFaint:
864  if (currentFainter->canFight()) { return Stage::WaitingFaintSwitch; }
865  else {
866  if (!state->enemy().canFight() && state->localPlayer().canFight()) { // player won
867  queueCommand({Command::NotifyBattleWinner, state->localPlayer().isHost()});
868  switch (battle->type) {
869  case Battle::Type::Trainer:
870  return Stage::TrainerDefeated;
871  case Battle::Type::Online:
872  return Stage::NetworkDefeated;
873  case Battle::Type::WildPeoplemon:
874  default:
875  return Stage::Completed;
876  }
877  }
878  else { // player lost
879  queueCommand({Command::NotifyBattleWinner, state->enemy().isHost()});
880  return Stage::Whiteout;
881  }
882  }
883  break;
884 
885  case Stage::XpAwardBegin:
886  xpAward =
887  currentFainter->activePeoplemon().base().xpYield(battle->type == Battle::Type::Trainer);
888  xpAward /= std::max(state->localPlayer().xpEarnerCount(), 1);
889  xpAwardRemaining = xpAward;
890  xpAwardIndex = state->localPlayer().getFirstXpEarner();
891  return xpAwardIndex >= 0 ? Stage::XpAwardPeoplemonBegin : Stage::CheckFaint;
892 
893  case Stage::BeforeRun: {
894  // TODO - should we allow "running" from online battles?
895  if (battle->type != Battle::Type::WildPeoplemon) { return Stage::RunFailed; }
896  ++runCount;
897  const int ps = state->localPlayer().activePeoplemon().currentStats().spd;
898  const int os = state->enemy().activePeoplemon().currentStats().spd;
899  const int odds = ((ps * 128 / os) + 30 * runCount) % 256;
900 
901 #ifdef PEOPLEMON_DEBUG
902  const bool alwaysRun = debug::DebugOverrides::get().alwaysRun ||
903  state->localPlayer().activePeoplemon().currentAbility() ==
904  pplmn::SpecialAbility::RunAway;
905 #else
906  const bool alwaysRun = state->localPlayer().activePeoplemon().currentAbility() ==
907  pplmn::SpecialAbility::RunAway;
908 #endif
909 
910  if (bl::util::Random::get<int>(0, 255) < odds || alwaysRun) { return Stage::Running; }
911  else {
912  return Stage::RunFailed;
913  }
914  } break;
915 
916  case Stage::UsingItem:
917  if (item::Item::getType(state->activeBattler().chosenItem()) == item::Type::Peopleball) {
918  if (state->activeBattler().chosenItem() == item::Id::CloneBall) {
919  if (battle->type != Battle::Type::WildPeoplemon &&
920  state->inactiveBattler().activePeoplemon().base().canClone()) {
921  return Stage::CloneBallThrown;
922  }
923  else {
924  return Stage::CloneBallFailed;
925  }
926  }
927 
928  if (battle->type == Battle::Type::WildPeoplemon) { return Stage::PeopleballThrown; }
929  else {
930  return Stage::PeopleballStealFailed;
931  }
932  }
933  return Stage::UsingItem;
934 
935  default:
936  return ns;
937  }
