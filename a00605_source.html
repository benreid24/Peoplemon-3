98 void Interaction::processConversationNode() {
99  if (interactingEntity == bl::entity::InvalidEntity) return;
100  if (currentConversation.finished()) {
101  owner.controllable().resetEntityLock(interactingEntity);
102  return;
103  }
104 
105  using E = file::Conversation::Node::Type;
106  const file::Conversation::Node& node = currentConversation.currentNode();
107  switch (node.getType()) {
108  case E::Talk:
109  owner.hud().displayMessage(node.message(), std::bind(&Interaction::continuePressed, this));
110  break;
111 
112  case E::Prompt: {
113  std::vector<std::string> choices;
114  choices.reserve(node.choices().size());
115  for (const auto& pair : node.choices()) choices.push_back(pair.first);
116  owner.hud().promptUser(node.message(),
117  choices,
118  std::bind(&Interaction::choiceMade, this, std::placeholders::_1));
119  } break;
120 
121  case E::GiveItem:
122  owner.player().bag().addItem(node.item().id);
123  if (node.item().afterPrompt) {
124  // TODO - add prefix to item metadata for a/the use
125  owner.hud().displayMessage("Got a " + item::Item::getName(node.item().id),
126  std::bind(&Interaction::continuePressed, this));
127  }
128  break;
129 
130  case E::TakeItem:
131  if (node.item().beforePrompt) {
132  owner.hud().promptUser(
133  "Hand over a " + item::Item::getName(node.item().id),
134  {"Yes", "No"},
135  std::bind(&Interaction::giveItemDecided, this, std::placeholders::_1));
136  }
137  else {
138  if (owner.player().bag().removeItem(currentConversation.currentNode().item().id)) {
139  currentConversation.notifyCheckPassed();
140  processConversationNode();
141  }
142  }
143  break;
144 
145  case E::GiveMoney:
146  owner.player().money() += node.money();
147  owner.hud().displayMessage("Received " + std::to_string(node.money()) + " monies",
148  std::bind(&Interaction::continuePressed, this));
149  break;
150 
151  case E::TakeMoney:
152  owner.hud().promptUser(
153  "Fork over " + std::to_string(node.money()) + " monies?",
154  {"Yes", "No"},
155  std::bind(&Interaction::giveMoneyDecided, this, std::placeholders::_1));
156  break;
157 
158  default:
159  BL_LOG_ERROR << "Invalid conversation node type " << node.getType() << ", terminating";
160  owner.controllable().resetEntityLock(interactingEntity);
161  interactingEntity = bl::entity::InvalidEntity;
162  break;
163  }
