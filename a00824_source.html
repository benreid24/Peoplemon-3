159 void Player::updateLantern(float dt) {
160  auto& lighting = owner.world().activeMap().lightingSystem();
161 
162  if (lighting.lightsAreOn()) {
163  if (lanternVariance != lanternTargetVariance) {
164  if (lanternVariance < lanternTargetVariance) {
165  lanternVariance += varianceConvergeRate * dt;
166  if (lanternVariance >= lanternTargetVariance) { startLanternVarianceHold(); }
167  }
168  else {
169  lanternVariance -= varianceConvergeRate * dt;
170  if (lanternVariance <= lanternTargetVariance) { startLanternVarianceHold(); }
171  }
172  }
173  else {
174  varianceSwitchTime -= dt;
175  if (varianceSwitchTime <= 0.f) { startLanternVarianceChange(); }
176  }
177 
178  lighting.updateLight(lantern, makeLight());
179  }
180  else {
181  lighting.removeLight(lantern);
182  lantern = map::LightingSystem::None;
183  }
184 }
