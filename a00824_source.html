117 void Player::updateLantern(float dt) {
118  auto& lighting = owner.world().activeMap().lightingSystem();
119 
120  if (lighting.lightsAreOn()) {
121  if (lanternVariance != lanternTargetVariance) {
122  if (lanternVariance < lanternTargetVariance) {
123  lanternVariance += varianceConvergeRate * dt;
124  if (lanternVariance >= lanternTargetVariance) { startLanternVarianceHold(); }
125  }
126  else {
127  lanternVariance -= varianceConvergeRate * dt;
128  if (lanternVariance <= lanternTargetVariance) { startLanternVarianceHold(); }
129  }
130  }
131  else {
132  varianceSwitchTime -= dt;
133  if (varianceSwitchTime <= 0.f) { startLanternVarianceChange(); }
134  }
135 
136  lighting.updateLight(lantern, makeLight());
137  }
138  else {
139  lighting.removeLight(lantern);
140  lantern = map::LightingSystem::None;
141  }
142 }
