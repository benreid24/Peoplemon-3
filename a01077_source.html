31 void shiftDownJumps(std::vector<Conversation::Node>& nodes, unsigned int i) {
32  const auto updateJump = [&nodes, i](std::uint32_t& jump) {
33  const auto isSimpleNext = [&nodes](unsigned int i) {
34  if (i >= nodes.size()) return false;
35  switch (nodes[i].getType()) {
36  case Conversation::Node::Talk:
37  case Conversation::Node::GiveMoney:
38  case Conversation::Node::GiveItem:
39  case Conversation::Node::RunScript:
40  case Conversation::Node::SetSaveFlag:
41  return true;
42 
43  default:
44  return false;
45  }
46  };
47 
48  if (jump == i) {
49  if (isSimpleNext(jump)) {
50  const std::uint32_t j = nodes[jump].next();
51  jump = j > i ? j - 1 : j;
52  }
53  else {
54  jump = nodes.size();
55  }
56  }
57  else if (jump > i)
58  --jump;
59  };
60 
61  for (unsigned int j = 0; j < nodes.size(); ++j) {
62  Conversation::Node& node = nodes[j];
63 
64  // Invalidate or shift down
65  updateJump(node.nextOnPass());
66  updateJump(node.nextOnReject());
67 
68  // Choices
69  if (node.getType() == Conversation::Node::Prompt) {
70  for (auto& pair : node.choices()) {
71  // Invalidate or shift down
72  updateJump(pair.second);
73  }
74  }
75  }
