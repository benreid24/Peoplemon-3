121 void Towns::refreshSpawns(std::uint16_t s) {
122  spawnSelect->clearOptions();
123  spawnSelect->addOption("None");
124 
125  std::vector<std::uint16_t> spawns;
126  spawns.reserve(map.spawns.size());
127  for (const auto& spawn : map.spawns) { spawns.emplace_back(spawn.first); }
128  std::sort(spawns.begin(), spawns.end());
129 
130  int opt = 0;
131  for (unsigned int i = 0; i < spawns.size(); ++i) {
132  spawnSelect->addOption(std::to_string(spawns[i]));
133  if (spawns[i] == s) { opt = i + 1; }
134  }
135  spawnSelect->setSelectedOption(opt);
136 }
