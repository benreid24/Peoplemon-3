102  case SwitchMapFadeout:
103  fadeTime += dt;

105  BL_LOG_INFO << "Switching to map: " << replacementMap << " spawn " << spawnId;
106  if (!systems.world().switchMaps(replacementMap, spawnId)) {
107  BL_LOG_ERROR << "Failed to switch maps";
