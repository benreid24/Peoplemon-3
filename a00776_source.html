196 void BagMenu::update(bl::engine::Engine& engine, float dt) {
197  systems.player().update();
198 
199  if (state == MenuState::Browsing) {
200  const core::component::Command input = inputDriver.mostRecentInput();
201 
202  if (actionOpen) {

232  activeMenu = &tmMenu;

235  else if (activeMenu == &keyMenu) {
236  activeMenu = &regularMenu;

240  activeMenu = &keyMenu;

245  break;
246 
247  default:
248  break;
249  }
250  }
251  }

253  slideOff += slideVel * dt;
