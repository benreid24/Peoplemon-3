253 bool CatchRegionWindow::validate() const {
254  if (rows.empty()) {
255  bl::dialog::tinyfd_messageBox(
256  "Error", "Region must have at least one Peoplemon", "ok", "error", 1);
257  return false;
258  }
259  if (nameEntry->getInput().empty()) {
260  bl::dialog::tinyfd_messageBox("Error", "Please enter a name", "ok", "error", 1);
261  return false;
262  }
263 
264  float sum = 0.f;
265  for (const Row& row : rows) {
266  if (row.idSelect->currentPeoplemon() == core::pplmn::Id::Unknown) {
267  bl::dialog::tinyfd_messageBox("Error", "Peoplemon cannot be blank", "ok", "error", 1);
268  return false;
269  }
270  if (!isnum(row.minLevel->getInput()) || !isnum(row.maxLevel->getInput())) {
271  bl::dialog::tinyfd_messageBox("Error", "Levels must be numbers", "ok", "error", 1);
272  return false;
273  }
274  const int mn = std::atoi(row.minLevel->getInput().c_str());
275  const int mx = std::atoi(row.maxLevel->getInput().c_str());
276  if (mn <= 0 || mx <= 0) {
277  bl::dialog::tinyfd_messageBox("Error", "Levels must be non-zero", "ok", "error", 1);
278  return false;
279  }
280  if (mn > 100 || mx > 100) {
281  bl::dialog::tinyfd_messageBox(
282  "Error", "Levels must be less than or equal to 100", "ok", "error", 1);
283  return false;
284  }
285  if (mx < mn) {
286  bl::dialog::tinyfd_messageBox(
287  "Error", "Min level must be below max level", "ok", "error", 1);
288  return false;
289  }
290  sum += row.freqSlider->getValue();
291  }
292 
293  if (std::abs(1.f - sum) > 0.001f) {
294  bl::dialog::tinyfd_messageBox("Error", "Frequencies must sum to 100%", "ok", "error", 1);
295  return false;
296  }
297 
298  return true;
299 }
