613 Value waitUntilTime(system::Systems& systems, SymbolTable& table, const std::vector<Value>& args) {
614  Function::validateArgs<Value::TNumeric, Value::TBool>("waitUntilTime", args);
615 
616  const system::Clock::Time now = systems.clock().now();
617  const system::Clock::Time then = parseTime(args[0]);
618  const unsigned int nts = now.hour * 60 + now.minute;
619  const unsigned int ets = then.hour * 60 + then.minute;
620  if (nts < ets || (nts != ets && args[1].deref().getAsBool())) {
621  bl::util::Waiter waiter;
622  const auto unlock = [&waiter]() { waiter.unblock(); };
623  ClockTrigger trigger(unlock, then);
624  bl::event::ClassGuard<event::TimeChange> guard(&trigger);
625  guard.subscribe(systems.engine().eventBus());
626  table.waitOn(waiter);
627  }
628 
629  return {};
630 }
